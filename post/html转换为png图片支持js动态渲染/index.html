#/bin/bash
 
#set -v -x

. param.sh

for i in "${ar[@]}" ; do
    b=($i) #此时b就相当于二维数组里面的一维数组了，然后可以再次遍历
    uigId=${b[0]}
    uId=${b[1]}
    goodsId=${b[2]}
    money=${b[3]}
    buaId=${b[4]}

    path="pdf/${buaId}/${uigId}_${uId}.pdf"
    if [ ! -f "$path" ]; then
        echo $path
        phantomjs rasterize.js "https://*/xwm/agreement/detail/new_debt_trans_person/debt/?uid=${uId}&uigId=${uigId}&goodsId=${goodsId}&money=${money}" "${path}"
    fi
done
